% ========== Example 3: Subjective (existing) ==========
\begin{questionboxappendix}{Examples of Subjective Q\&A}
\label{box:subj:blockchain_analysis}
\textbf{Question Type:} Open-ended Analysis \\
\textbf{Scenario:} Explain consensus mechanisms in blockchain \\
\textbf{Instructions:} Analyze the principles, advantages, disadvantages, and applicable scenarios of different types of consensus mechanisms (such as PoW, PoS, DPoS, etc.). \\
\textbf{Factors to Consider:} 
\begin{itemize}[leftmargin=*]
    \item Technical principles of various consensus mechanisms
    \item Energy consumption and sustainability
    \item Security and degree of decentralization
    \item Scalability and transaction throughput
\end{itemize}
\textbf{Keywords:} Proof of Work, Proof of Stake, Delegated Proof of Stake, Byzantine Fault Tolerance \\
\textbf{Expected Insights:} 
\begin{itemize}[leftmargin=*]
    \item PoW provides strong security but high energy consumption
    \item PoS reduces energy consumption but faces the rich-get-richer problem
    \item DPoS improves efficiency but reduces the degree of decentralization
    \item Consensus mechanism selection involves trade-offs between security, scalability, and decentralization
\end{itemize}
\textbf{Scoring Criteria (Total: 10 points):}
\begin{itemize}[leftmargin=*]
    \item Technical principles explanation (4 points): Accurately describe the working principles of at least 3 major consensus mechanisms; explain security assumptions and attack protections
    \item Advantages and disadvantages analysis (3 points): Comprehensively compare advantages and disadvantages; analyze performance in energy efficiency, decentralization, and throughput
    \item Application scenario recommendations (3 points): Suggest appropriate mechanisms for different blockchain projects, considering requirements, scale, and security needs
\end{itemize}
\textbf{Domain:} Blockchain Fundamentals \\
\textbf{Sub-topic:} Consensus Mechanisms \\
\textbf{Difficulty:} Medium \\
\textbf{Ability Type:} Critical Analysis \\
\textbf{Source URL:} \url{https://ethereum.org/en/developers/docs/consensus-mechanisms/} \\
\textbf{Review Status:} Expert\_Verified \\
\textbf{Expert:} Frank Li
\end{questionboxappendix}

% ========== Example 4: Subjective (different type: Market Reasoning) ==========
\begin{questionboxappendix}{Examples of Subjective Q\&A (cont.)}
\label{box:subj:market_reasoning}
\textbf{Question Type:} Market Reasoning \\
\textbf{Scenario:} A well-known cryptocurrency exchange has been hacked, with hackers successfully stealing crypto assets valued at \$200 million. \\
\textbf{Instructions:} Analyze the potential impact of this event on the entire cryptocurrency market. \\
\textbf{Factors to Consider:} 
\begin{itemize}[leftmargin=*]
    \item Exchange reputation and market position
    \item Types and amount of stolen assets
    \item Exchange's response measures
    \item Potential reactions from regulatory agencies
\end{itemize}
\textbf{Keywords:} Hacking attack, Market confidence, Security vulnerabilities, Regulatory scrutiny \\
\textbf{Expected Insights:} 
\begin{itemize}[leftmargin=*]
    \item Major exchange security incidents typically trigger short-term market panic
    \item The exchange's response measures are key to restoring market confidence
    \item Such events may lead to stricter industry regulation
    \item Analysis of how similar historical events impacted the market
\end{itemize}
\textbf{Scoring Criteria (Total: 10 points):}
\begin{itemize}[leftmargin=*]
    \item Market impact analysis (4 points): Analyze the potential effect on short-term and long-term market sentiment; evaluate the potential impact on prices of different types of crypto assets
    \item Security vulnerability assessment (3 points): Explore potential systemic security issues exposed by the attack; discuss security measures that exchanges and users can implement
    \item Regulatory impact prediction (3 points): Predict possible reactions from regulatory agencies; assess the potential impact of additional regulation on industry development
\end{itemize}
\textbf{Domain:} Smart Contract Security \\
\textbf{Sub-topic:} Exchange Security \\
\textbf{Difficulty:} Medium \\
\textbf{Ability Type:} Risk Assessment \\
\textbf{Source URL:} \url{https://consensys.github.io/smart-contract-best-practices/} \\
\textbf{Review Status:} Expert\_Verified \\
\textbf{Expert:} Zhiyu Zhang
\end{questionboxappendix}
